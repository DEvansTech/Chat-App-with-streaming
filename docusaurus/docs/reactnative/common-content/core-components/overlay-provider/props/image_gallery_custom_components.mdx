- [ImageGalleryFooter](#)

{
 footer: {
   ShareIcon: CustomShareIconComponent
 },
 grid: {
   avatarComponent: CustomAvatarComponent
 },
 gridHandle: {
   centerComponent: CustomCenterComponent
 },
 header: {
   CloseIcon: CustomCloseButtonComponent
 },
}
